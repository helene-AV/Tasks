30 inline void fillQC(const std::vector<Task *> & tasks, int nrVars, Eigen::MatrixXd & Q, Eigen::VectorXd & C)
31 {
32  for(std::size_t i = 0; i < tasks.size(); ++i)
33  {
34  const Eigen::MatrixXd & Qi = tasks[i]->Q();
35  const Eigen::VectorXd & Ci = tasks[i]->C();
36  std::pair<int, int> b = tasks[i]->begin();
37 
38  int r = static_cast<int>(Qi.rows());
39  int c = static_cast<int>(Qi.cols());
40 
41  Q.block(b.first, b.second, r, c) += tasks[i]->weight() * Qi;
42  C.segment(b.first, r) += tasks[i]->weight() * Ci;
43  }
44 
45  // try to transform Q_ to a positive matrix
46  // we just add a small value to the diagonal since
47  // the first necessary condition is to have
48  // Q_(i,i) > 0
49  // may be we can try to check the second
50  // condition in a near futur
51  // Q_(i,i) + Q_(j,j) > 2·Q_(i,j) for i≠j
52  for(int i = 0; i < nrVars; ++i)
53  {
54  if(std::abs(Q(i, i)) < DIAG_CONSTANT)
55  {
56  Q(i, i) += DIAG_CONSTANT;
57  }
58  }
59 }

307 inline void reduceBound(const Eigen::VectorXd & XLFull,
308  Eigen::VectorXd & XL,
309  const Eigen::VectorXd & XUFull,
310  Eigen::VectorXd & XU,
311  const std::vector<int> & fullToReduced,
312  const std::vector<int> & reducedToFull,
313  const std::vector<std::tuple<int, int, double>> & dependencies)
314 {
315  for(size_t i = 0; i < static_cast<size_t>(XL.rows()); ++i)
316  {
317  XL(static_cast<Eigen::DenseIndex>(i)) = XLFull(reducedToFull[i]);
318  XU(static_cast<Eigen::DenseIndex>(i)) = XUFull(reducedToFull[i]);
319  }
320  for(const auto & d : dependencies)
321  {
322  const int & primaryFullI = std::get<0>(d);
323  const int & primaryReducedI = fullToReduced[static_cast<size_t>(primaryFullI)];
324  const int & replicaFullI = std::get<1>(d);
325  const double & alpha = std::get<2>(d);
326  if(alpha != 0)
327  {
329  if(alpha < 0)
330  {
331  XL(primaryReducedI) = std::max(XL(primaryReducedI), XUFull(replicaFullI) / alpha);
332  XU(primaryReducedI) = std::min(XU(primaryReducedI), XLFull(replicaFullI) / alpha);
333  }
334  else
335  {
336  XL(primaryReducedI) = std::max(XL(primaryReducedI), XLFull(replicaFullI) / alpha);
337  XU(primaryReducedI) = std::min(XU(primaryReducedI), XUFull(replicaFullI) / alpha);
338  }
339  }
340  }
341 }
